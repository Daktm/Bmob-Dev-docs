<static> {Class} Bmob.Collection.extend(instanceProps, classProps)
Creates a new subclass of Bmob.Collection. For example,
  var MyCollection = Bmob.Collection.extend({
    // Instance properties

    model: MyClass,
    query: MyQuery,

    getFirst: function() {
      return this.at(0);
    }
  }, {
    // Class properties

    makeOne: function() {
      return new MyCollection();
    }
  });

  var collection = new MyCollection();
